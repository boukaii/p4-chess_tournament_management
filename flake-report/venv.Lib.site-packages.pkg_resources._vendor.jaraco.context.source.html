58 def infer_compression(url):
59 """
60 Given a URL or filename, infer the compression code for tar.
61  
62 >>> infer_compression('http://foo/bar.tar.gz')
63 'z'
64 >>> infer_compression('http://foo/bar.tgz')
65 'z'
66 >>> infer_compression('file.bz')
67 'j'
68 >>> infer_compression('file.xz')
69 'J'
