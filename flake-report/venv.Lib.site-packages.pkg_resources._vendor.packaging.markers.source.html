48 Normalize extra values.

103 def _eval_op(lhs: str, op: Op, rhs: str) -> bool:
104 try:
105 spec = Specifier("".join([op.serialize(), rhs]))
106 except InvalidSpecifier:
107 pass
108 else:
109 return spec.contains(lhs, prereleases=True)
110  
111 oper: Optional[Operator] = _operators.get(op.serialize())
112 if oper is None:
113 raise UndefinedComparison(f"Undefined {op!r} on {lhs!r} and {rhs!r}.")
114  
115 return oper(lhs, rhs)

161 version = "{0.major}.{0.minor}.{0.micro}".format(info)
162 kind = info.releaselevel
163 if kind != "final":
164 version += kind[0] + str(info.serial)
